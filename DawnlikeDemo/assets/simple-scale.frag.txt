#ifdef GL_ES
#define LOWP lowp
#ifdef GL_FRAGMENT_PRECISION_HIGH
precision highp float;
#else
precision mediump float;
#endif
#define COMPAT_PRECISION mediump
#else
#define COMPAT_PRECISION
#define LOWP
#endif

uniform COMPAT_PRECISION vec2 TextureSize;
uniform COMPAT_PRECISION vec2 Scale;
uniform sampler2D u_texture;
varying LOWP vec4 COL0;
varying vec2 v_texCoords;

void main()
{
    vec2 pixel = v_texCoords * TextureSize / (Scale * 0.25);
    vec2 uv = floor(pixel) + 0.5;
    uv += 0.5 - clamp((1.0 - fract(pixel)) * Scale * 0.25, 0.0, 1.0);
    gl_FragColor = texture2D(u_texture, uv * 0.25 * Scale / TextureSize) * COL0;
}
